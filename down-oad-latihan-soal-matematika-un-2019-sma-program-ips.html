1. soal un tdi yan step up dan step down apa ya?


intinya
Step up =
Np < Ns , Vp < Vs , Ip > Is 
Step down =
Np > Ns , Vp > Vs , Ip < Is
V = tegangan
N = jumlah lilitan
I = kuat arus
p = primer
s =sekunder

Penjelasan dengan langkah-langkah:

sudut yang berada di 2garis berpotongan besarnya 90derajat, AOB membentuk segitiga siku2 di O, jumlah sudut pada segitiga adalah 180derajat. maka

90 + OAB +OAD= 180

15x+30x = 180-90

45x = 90

x = 90 : 45

x=2

jadi sudut OAD = 30(2) = 60derajat

// A  B  C  D  E  F  G  DP

 {1, 1, 1, 1, 0, 0, 1, 0}, // 3

 {0, 1, 1, 0, 0, 1, 1, 0}, // 4

 {1, 0, 1, 1, 0, 1, 1, 0}, // 5

 {1, 0, 1, 1, 1, 1, 1, 0}, // 6

 {1, 1, 1, 0, 0, 0, 0, 0}, // 7

 {1, 1, 1, 1, 1, 1, 1, 0}, // 8

 {1, 1, 1, 1, 0, 1, 1, 0}  // 9

};

void setup() {

 for (byte indeks = 0; indeks <8; indeks++) {

   pinMode(pin[indeks], OUTPUT);

   digitalWrite(pin[indeks], HIGH);

 }

 

 Serial.begin(9600);

}

void loop() {

 int bil, indeks;

 

 for (bil = 9; bil >= 0; bil--) {

   Serial.println(bil);

   // Matikan semua segmen

   for (indeks = 0; indeks < 8; indeks++)

       digitalWrite(pin[indeks], HIGH);

       

   // Tampilkan angka

   for (indeks = 0; indeks < 8; indeks++)

     if (angka[bil][indeks] == 1)

       digitalWrite(pin[indeks], LOW);

       

   delay(500);    // Tunda

 }  

}
