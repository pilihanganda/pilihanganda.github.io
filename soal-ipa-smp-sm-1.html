Penjelasan dengan langkah-langkah:

no. 1

y = 2x² - 4x + 3

y = x² - 3x + 5

2x² - 4x + 3 = x² - 3x + 5

2x² - x² - 4x + 3x + 3 - 5 = 0

x² - x - 2 = 0

(x + 1)(x - 2) = 0

x = - 1 atau x = 2

HP = { - 1, 2 }

no. 5

y = x² + x - 2

y = 2x² - 3x + 1

2x² - 3x + 1 = x² + x - 2

2x² - x² - 3x - x + 1 + 2 = 0

x² - 4x + 3 = 0

(x - 1)(x - 3) = 0

x = 1 atau x = 3

HP = { 1, 3 }
